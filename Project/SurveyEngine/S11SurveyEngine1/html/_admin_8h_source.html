540 void readFromBin(string file) {
541 surveys.clear(); //Clear vector first to read entire file contents to it
542 fstream surveyDB(file, ios::in | ios::binary);
543 if (!surveyDB) {
544 cerr << "ERROR: Unable to Open File for Reading." << endl;
545 return;
546 }
547
548 //Use cursor to get size of file to read entire file
549 surveyDB.seekg(0, ios::end);
550 int size = surveyDB.tellg();
551 surveyDB.seekg(0, ios::beg);
552
553 //Read in Survey information until cursor reaches the size of the file
554 while (surveyDB.tellg() < size) {
555 //Read title and number of questions from binary file
556 char title[TSIZE];
557 bool status;
558 int numQs;
559 surveyDB.read(reinterpret_cast<char *> (title), TSIZE);
560 surveyDB.read(reinterpret_cast<char *> (&status), sizeof (bool));
561 surveyDB.read(reinterpret_cast<char *> (&numQs), sizeof (int));
562 vector<Question *> questions;
563 //Loop through each question in the binary file and read
564 for (int i = 0; i < numQs; i++) {
565 char q[QRSIZE];
566 int type, numResp;
567 vector<string> options;
568
569 surveyDB.read(reinterpret_cast<char *> (q), sizeof (q));
570 surveyDB.read(reinterpret_cast<char *> (&type), sizeof (int));
571 surveyDB.read(reinterpret_cast<char *> (&numResp), sizeof (int));
572 //Loop through each response option in the binary file and read
573 for (int j = 0; j < numResp; j++) {
574 int len;
575 string option;
576
577 surveyDB.read(reinterpret_cast<char *> (&len), sizeof (int));
578 option.resize(len);
579 surveyDB.read(&option[0], len);
580 options.push_back(option);
581 }
582 // Create new Question object and set its properties
583 Question *newQ = new Question();
584 newQ->setQ(q);
585 newQ->setType(type);
586 newQ->setNumResp(numResp);
587 newQ->setRespOptions(options);
588 questions.push_back(newQ);
589 }
590 //Create new Survey object,set properties, and add it to vector
591 Survey *survey = new Survey();
592 survey->setTitle(title);
593 survey->setStatus(status);
594 survey->setNumQs(numQs);
595 survey->setQs(questions);
596 surveys.push_back(survey);
597 }
598 surveyDB.close();
599 }
