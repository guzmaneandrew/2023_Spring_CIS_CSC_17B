556 void readFromBin(string file, vector<Survey *>& srvs) {
557 srvs.clear(); //Clear vector first to read entire file contents to it
558 fstream surveyDB(file, ios::in | ios::binary);
559 if (!surveyDB) {
560 return;
561 }
562
563 //Use cursor to get size of file to read entire file
564 surveyDB.seekg(0, ios::end);
565 int size = surveyDB.tellg();
566 surveyDB.seekg(0, ios::beg);
567
568 //Read in Survey information until cursor reaches the size of the file
569 while (surveyDB.tellg() < size) {
570 //Read title and number of questions from binary file
571 char title[TSIZE];
572 bool status;
573 int numQs;
574 surveyDB.read(reinterpret_cast<char *> (title), TSIZE);
575 surveyDB.read(reinterpret_cast<char *> (&status), sizeof (bool));
576 surveyDB.read(reinterpret_cast<char *> (&numQs), sizeof (int));
577 vector<Question *> questions;
578 //Loop through each question in the binary file and read
579 for (int i = 0; i < numQs; i++) {
580 char q[QRSIZE];
581 int type, numResp;
582 vector<string> options;
583
584 surveyDB.read(reinterpret_cast<char *> (q), sizeof (q));
585 surveyDB.read(reinterpret_cast<char *> (&type), sizeof (int));
586 surveyDB.read(reinterpret_cast<char *> (&numResp), sizeof (int));
587 //Loop through each response option in the binary file and read
588 for (int j = 0; j < numResp; j++) {
589
590 int len;
591 string option;
592 surveyDB.read(reinterpret_cast<char *> (&len), sizeof (int));
593 option.resize(len);
594 surveyDB.read(&option[0], len);
595 options.push_back(option);
596 }
597
598 // Create new Question object and set its properties
599 Question *newQ = new Question();
600 newQ->setQ(q);
601 newQ->setType(type);
602 newQ->setNumResp(numResp);
603 newQ->setRespOptions(options);
604 questions.push_back(newQ);
605 }
606
607 //Create new Survey object,set properties, and add it to vector
608 Survey *survey = new Survey();
609 survey->setTitle(title);
610 survey->setStatus(status);
611 survey->setNumQs(numQs);
612 survey->setQs(questions);
613 srvs.push_back(survey);
614 }
615 surveyDB.close();
616 }
